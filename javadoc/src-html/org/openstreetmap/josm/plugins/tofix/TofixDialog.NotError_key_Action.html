286    public void fillCombo() {
287        listStringsForCombo.clear();
288        listStringsForCombo.add(tr("Select a project ..."));
289        if (Status.isInternetReachable()) {
290            projectsList = listProjectController.getListProjects();
291                for (int i = 0; i < projectsList.getProjects().size(); i++) {
292                    listStringsForCombo.add(projectsList.getProjects().get(i).getName());
293                }
294                jcomboBox.setModel(new DefaultComboBoxModel<>());
295                jcomboBox.setModel(new DefaultComboBoxModel<>(listStringsForCombo.toArray(new String[]{})));            
296        } else {
297            skipButton.setEnabled(false);
298            fixedButton.setEnabled(false);
299            noterrorButton.setEnabled(false);
300        }
301    }

303    public void setToken() {
304        if (Status.serverStatus()) {
305            oauth.setVisible(true);
306            if (oauth.getTofixToken() != null && !oauth.getTofixToken().equals("")) {
307                if (Config.preferences(Config.GET, new String[]{"tofix-server.token"}, Config.getPluginPreferencesFile().getAbsolutePath()) != null) {
308                    Config.preferences(Config.UPDATE, new String[]{"tofix-server.token", oauth.getTofixToken()}, Config.getPluginPreferencesFile().getAbsolutePath());
309                } else {
310                    Config.preferences(Config.ADD, new String[]{"tofix-server.token", oauth.getTofixToken()}, Config.getPluginPreferencesFile().getAbsolutePath());
311                }
312                fillCombo();
313            } else {
314                fillCombo();
315            }
316        } else {
317            JOptionPane.showMessageDialog(Main.parent, tr("API did not response! ") + Config.getHOST());
318        }
319
320    }
321    private void setHost() {
322        host.setVisible(true);
323        if (host.getHost() != null && host.getHost().equals("")) {
324            fillCombo();
325        }

328// Event select a project, it automatic will get a item and display
329    @Override
330    public void actionPerformed(ActionEvent e) {
331        start();
332        JComboBox<?> cb = (JComboBox<?>) e.getSource();
333        if (cb.getSelectedIndex() != 0) {
334            mainAccessToProject.setProject_id(projectsList.getProjects().get(cb.getSelectedIndex() - 1).getId());
335            mainAccessToProject.setProject_name(projectsList.getProjects().get(cb.getSelectedIndex() - 1).getName());
336            project = projectsList.getProjects().get(cb.getSelectedIndex() - 1);
337            deleteLayer();
338            getNewItem();
339            skipButton.setEnabled(true);
340            fixedButton.setEnabled(true);
341            noterrorButton.setEnabled(true);
342        } else {
343            skipButton.setEnabled(false);
344            fixedButton.setEnabled(false);
345            noterrorButton.setEnabled(false);
346        }
347    }

349    private void loadOAuthInfo() {
350        oauth.setUserInfo(Config.getUserName(), Config.getPassword(), Config.getTOKEN());
351        
352        if (oauth.getTofixToken() != null && !oauth.getTofixToken().equals("")) { 
353            if(Status.serverStatus()){
354              fillCombo();
355            }           
356            if(Config.isDefaultToken(oauth.getTofixToken())){
357                jCheckBoxToken.setSelected(true);
358            }else{
359                jCheckBoxToken.setSelected(false);
360            }
361        }
362        host.setHost(Config.getHOST());
363        if (host.getHost() != null && !host.getHost().equals("")) {            
364            if(Config.isDefaultAPI(host.getHost())){
365                jCheckBoxSetNewAPI.setSelected(true);
366            }else{
367                jCheckBoxSetNewAPI.setSelected(false);
368            }
369        }
370    }

398    private void eventFixed(ActionEvent e) {
399
400        if (!MainApplication.getLayerManager().getActiveLayer().isSavable() || !MainApplication.getLayerManager().getEditDataSet().isModified()) {
401            new Notification(tr("No change to upload!")).show();
402            //Be sure you mark as fixed
403            fixed();
404        } else if (new Bounds(MainApplication.getLayerManager().getEditDataSet().getDataSourceArea().getBounds()).getArea() < 30) {
405            validator = false;
406            UploadDialog.getUploadDialog().addComponentListener(new ComponentAdapter() {
407                @Override
408                public void componentShown(ComponentEvent e) {
409                    validator = true;
410                }
411            });
412            DataSet data = MainApplication.getLayerManager().getEditLayer().data;
413            project.getChangesetComment();
414            //Set up the changeset comment and the source
415            if (!project.getChangesetComment().equals("none")) {
416                data.getChangeSetTags().put("comment", project.getChangesetComment());
417            }
418            data.getChangeSetTags().put("source", MainApplication.getMap().mapView.getLayerInformationForSourceTag());
419            new UploadAction().actionPerformed(e);
420
421            if (validator && !UploadDialog.getUploadDialog().isCanceled() && UploadDialog.getUploadDialog().getChangeset().isNew()) {
422                fixed();
423                deleteLayer();
424            }
425        } else {
426            new Notification(tr("The bounding box of the edited layer is too big.")).show();
427        }
428    }

431    private void getNewItem() {
432        itemController.setAccessToProject(mainAccessToProject);
433        item = itemController.getItem();
434        switch (item.getStatusServer()) {
435            case 200:
436                mainAccessToProject.setAccess(true); //This atribute to access to  the actions
437                mainAccessToProject = tofixProject.work(item, mainAccessToProject, zise, isCheckedDownloadOSMData, isCheckedEditableData);
438                edit();
439                break;
440            case 410:
441                JOptionPane.showMessageDialog(Main.parent, tr("There are no more items on this Project or Area!"), tr("Warning"), JOptionPane.WARNING_MESSAGE);
442                mainAccessToProject.setAccess(false);
443                break;
444            default:
445                mainAccessToProject.setAccess(false);
446                new Notification(tr("Something went wrong, try again")).show();
447        }
448    }

479    public void deleteLayer() {
480        if (needDeleteLayer) {
481            OsmDataLayer editLayer = MainApplication.getLayerManager().getEditLayer();
482            if (editLayer != null) {
483                editLayer.data.clear();
484                MainApplication.getLayerManager().removeLayer(editLayer);
485            }
486        }
