001package org.openstreetmap.josm.plugins.tofix.controller;
002
003import java.io.StringReader;
004import java.util.logging.Level;
005import java.util.logging.Logger;
006import javax.json.Json;
007import javax.json.JsonArray;
008import javax.json.JsonObject;
009import javax.json.JsonReader;
010import org.openstreetmap.josm.plugins.tofix.bean.AccessToProject;
011import org.openstreetmap.josm.plugins.tofix.bean.ItemBean;
012import org.openstreetmap.josm.plugins.tofix.util.Request;
013import org.openstreetmap.josm.plugins.tofix.util.Util;
014
015/**
016 *
017 * @author ruben
018 */
019public class ItemController {
020
021    ItemBean item = new ItemBean();
022    JsonArray relation;
023    AccessToProject accessToProject;
024
025    public AccessToProject getAccessToProject() {
026        return accessToProject;
027    }
028
029    public void setAccessToProject(AccessToProject accessToProject) {
030        this.accessToProject = accessToProject;
031    }
032
033    public JsonArray getRelation() {
034        return relation;
035    }
036
037    public void setRelation(JsonArray relation) {
038        this.relation = relation;
039    }
040
041    public ItemBean getItem() {
042        try {
043            //Get item as String
044            String itemString = Request.sendGET(accessToProject.getProject_url());
045            JsonReader reader = Json.createReader(new StringReader(itemString));
046            JsonArray arrayItems = reader.readArray();
047            //Check if the array hays items
048            if (arrayItems.size() > 0) {
049                JsonObject itemObject = arrayItems.getJsonObject(0);
050                item.setId(itemObject.getString("id"));
051                item.setProject_id(itemObject.getString("project_id"));
052                item.setFeatureCollection(itemObject.getJsonObject("featureCollection"));
053                item.setStatusServer(200);
054
055            } else {
056                item.setStatusServer(410);
057            }
058            reader.close();
059        } catch (Exception ex) {
060            item.setStatusServer(502);
061            Logger.getLogger(ItemController.class.getName()).log(Level.SEVERE, null, ex);
062        }
063        return item;
064    }
065}
