001package org.openstreetmap.josm.plugins.tofix.controller;
002
003import java.io.IOException;
004import java.util.logging.Level;
005import java.util.logging.Logger;
006import javax.json.Json;
007import javax.json.JsonObject;
008import javax.json.JsonObjectBuilder;
009import org.openstreetmap.josm.plugins.tofix.bean.ItemBean;
010import org.openstreetmap.josm.plugins.tofix.util.Config;
011import org.openstreetmap.josm.plugins.tofix.util.Request;
012
013public class ItemTrackController {
014
015    public void lockItem(ItemBean item, String lock) {
016        //Create the URL
017        String url = Config.getHOST() + "/" + Config.API_VERSION + "/projects/" + item.getProject_id() + "/items/" + item.getId();
018        //Create data(json)to looked the item
019        JsonObjectBuilder trackBeanBuilder = Json.createObjectBuilder();
020
021        trackBeanBuilder
022                .add("lock", lock);
023
024        JsonObjectBuilder metadataBuilder = Json.createObjectBuilder();
025        metadataBuilder.add("editor", "JOSM");
026        trackBeanBuilder.add("metadata", metadataBuilder);
027        JsonObject track_edit = trackBeanBuilder.build();
028        try {
029            Request.sendPUT_Json(url, track_edit.toString());
030        } catch (IOException ex) {
031            Logger.getLogger(ItemTrackController.class.getName()).log(Level.SEVERE, null, ex);
032        }
033    }
034
035    public void updateStatusItem(ItemBean item, String status) {
036        //Create the URL
037        String url = Config.getHOST() + "/" + Config.API_VERSION + "/projects/" + item.getProject_id() + "/items/" + item.getId();
038        //Create data(json)to looked the item
039        JsonObjectBuilder trackBeanBuilder = Json.createObjectBuilder();
040
041        trackBeanBuilder
042                .add("status", status);
043
044        JsonObjectBuilder metadataBuilder = Json.createObjectBuilder();
045        metadataBuilder.add("editor", "JOSM");
046        trackBeanBuilder.add("metadata", metadataBuilder);
047        JsonObject track_edit = trackBeanBuilder.build();
048        try {
049            Request.sendPUT_Json(url, track_edit.toString());
050        } catch (IOException ex) {
051            Logger.getLogger(ItemTrackController.class.getName()).log(Level.SEVERE, null, ex);
052        }
053    }
054}
