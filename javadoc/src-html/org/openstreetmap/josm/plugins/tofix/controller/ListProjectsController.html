001package org.openstreetmap.josm.plugins.tofix.controller;
002
003import java.io.IOException;
004import java.io.StringReader;
005import java.text.Collator;
006import java.util.Collections;
007import java.util.LinkedList;
008import java.util.List;
009import java.util.logging.Level;
010import java.util.logging.Logger;
011
012import javax.json.Json;
013import javax.json.JsonArray;
014import javax.json.JsonObject;
015import javax.json.JsonReader;
016import javax.json.JsonValue;
017import javax.json.stream.JsonParsingException;
018
019import org.openstreetmap.josm.plugins.tofix.bean.ListProjectBean;
020import org.openstreetmap.josm.plugins.tofix.bean.ProjectBean;
021import org.openstreetmap.josm.plugins.tofix.util.Config;
022import org.openstreetmap.josm.plugins.tofix.util.Request;
023
024/**
025 *
026 * @author ruben
027 */
028public class ListProjectsController {
029
030    private final ListProjectBean listProjectsBean = new ListProjectBean();
031
032    public String getUrl(){
033        return Config.getHOST() + "/" + Config.API_VERSION + "/projects";
034    }
035    
036    public ListProjectBean getListProjects() {
037        List<ProjectBean> projects = new LinkedList<>();
038        try (JsonReader jsonReader = Json.createReader(new StringReader(Request.sendGET(getUrl())))) {
039            JsonArray jsonArray = jsonReader.readArray();
040            for (JsonValue value : jsonArray) {
041                ProjectBean projectBean = new ProjectBean();
042                try (JsonReader jsonReader2 = Json.createReader(new StringReader(value.toString()))) {
043                    JsonObject jsonProject = jsonReader2.readObject();
044                    projectBean.setId(jsonProject.getString("id"));
045                    projectBean.setName(jsonProject.getString("name"));
046                    projectBean.setMetadata(jsonProject.getJsonObject("metadata"));
047//                    projectBean.setQuadkey_set_id(jsonProject.getString("quadkey_set_id")); //FIXME
048                }
049                projects.add(projectBean);
050            }
051        } catch (IOException | NullPointerException | JsonParsingException ex) {
052            Logger.getLogger(ListProjectsController.class.getName()).log(Level.SEVERE, null, ex);
053        }
054        Collections.sort(projects, (o1, o2) -> Collator.getInstance().compare(o1.getName(), o2.getName()));
055        listProjectsBean.setProjects(projects);
056        return listProjectsBean;
057    }    
058}
