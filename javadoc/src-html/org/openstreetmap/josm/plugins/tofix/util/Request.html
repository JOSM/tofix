001package org.openstreetmap.josm.plugins.tofix.util;
002
003import java.io.IOException;
004import java.net.URL;
005import java.nio.charset.StandardCharsets;
006import org.openstreetmap.josm.tools.HttpClient;
007import org.openstreetmap.josm.tools.HttpClient.Response;
008
009/**
010 *
011 * @author ruben
012 */
013public class Request {
014    
015    public static void sendPOST_Json(String url, String object) throws IOException {
016        HttpClient.create(new URL(url), "POST")
017                .setHeader("Content-Type", "application/json")
018                .setHeader("Authorization", Config.getTOKEN())
019                .setAccept("application/json")
020                .setRequestBody(object.getBytes(StandardCharsets.UTF_8))
021                .connect().disconnect();
022    }
023
024    public static void sendPUT_Json(String url, String object) throws IOException {
025        HttpClient.create(new URL(url), "PUT")
026                .setHeader("Content-Type", "application/json")
027                .setHeader("Authorization", Config.getTOKEN())
028                .setAccept("application/json")
029                .setRequestBody(object.getBytes(StandardCharsets.UTF_8))
030                .connect().disconnect();
031    }
032    
033    public static String sendGET(String url) throws IOException {
034        Response response = HttpClient.create(new URL(url))
035                .setHeader("Authorization",Config.getTOKEN())
036                .connect();
037        String result = response.fetchContent();
038        response.disconnect();
039        return result;
040    }
041}
