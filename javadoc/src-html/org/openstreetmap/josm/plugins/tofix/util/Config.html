123    public static Object preferences(int op, String[] parms, String... path) {
124        if (path != null && path.length > 0) {
125            return preferences(path[0], op, parms);
126        } else {
127            return preferences(Config.getPreferencesFile().getAbsolutePath(), op, parms);
128        }
129    }
130
131    private static Object preferences(String path, int op, String[] parms) {
132        DocumentBuilder dBuilder;
133        Object r = null;
134        try {
135            dBuilder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
136            Document doc = dBuilder.parse(new File(path));
137            doc.getDocumentElement().normalize();
138
139            switch (op) {
140                case GET:
141                    r = get(doc, parms[0]);
142                    break;
143                case ADD:
144                    r = add(doc, parms[0], parms[1]);
145                    break;
146                case UPDATE:
147                    r = update(doc, parms[0], parms[1]);
148                    break;
149                case REMOVE:
150                    r = remove(doc, parms[0]);
151                    break;
152            }
153            doc.getDocumentElement().normalize();
154            Transformer transformer = TransformerFactory.newInstance().newTransformer();
155            DOMSource source = new DOMSource(doc);
156            StreamResult result = new StreamResult(new File(path));
157            transformer.setOutputProperty(OutputKeys.INDENT, "yes");
158            transformer.transform(source, result);
159
160        } catch (Exception e) {
161            Logging.error(e);
162        }
163        return r;
164    }

166    private static String get(Document doc, String key) throws Exception {
167        NodeList nList = doc.getElementsByTagName("tag");
168        for (int i = 0; i < nList.getLength(); i++) {
169            Node nNode = nList.item(i);
170            if (nNode.getNodeType() == Node.ELEMENT_NODE) {
171                Element eElement = (Element) nNode;
172                if (eElement.getAttribute("key").equals(key)) {
173                    return eElement.getAttribute("value");
174                }
175            }
176        }
177        return null;
178    }
179
180    private static String add(Document doc, String key, String value) {
181        NodeList p = doc.getElementsByTagName("preferences");
182        Element e1;
183        for (int i = 0; i < p.getLength(); i++) {
184            e1 = (Element) p.item(i);
185            Element e = doc.createElement("tag");
186            e.setAttribute("key", key);
187            e.setAttribute("value", value);
188            e1.appendChild(e);
189        }
190        return value;
191    }
192
193    private static String update(Document doc, String key, String value) {
194        NodeList employees = doc.getElementsByTagName("tag");
195        Element e = null;
196        for (int i = 0; i < employees.getLength(); i++) {
197            e = (Element) employees.item(i);
198            if (e.getNodeType() == Node.ELEMENT_NODE) {
199                if (e.getAttribute("key").equals(key)) {
200                    e.setAttribute("value", value);
201                    return "Success";
202                }
203            }
204        }
205        return "Success";
206    }
207    
208    private static String remove(Document doc, String key) throws Exception {
209        NodeList nList = doc.getElementsByTagName("preferences");
210        NodeList nListCh = nList.item(0).getChildNodes();
211        for (int i = 0; i < nListCh.getLength(); i++) {
212            
213            Node nNode = nListCh.item(i);
214            if (nNode.getNodeType() == Node.ELEMENT_NODE) {
215                Element eElement = (Element) nNode;
216                if (eElement.getAttribute("key").equals(key)) {
217                    nList.item(0).removeChild(nNode);
218                    return "Success";
219                }
220            }
221        }
222        return null;
223    }
