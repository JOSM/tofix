026/**
027 *
028 * @author ruben
029 */
030public class Download {
031
032    public static void download(Bounds bounds, final Long osm_obj_id, double downloadSize) {
033        DownloadOsmTask task = new DownloadOsmTask();
034        ProgressMonitor monitor = null;
035        //Fix bbox to download
036        if (bounds.getArea() == 0) {
037            bounds = new Bounds(new BBox(bounds.getCenter().getX(), bounds.getCenter().getY(), downloadSize).toRectangle());
038        } else if (bounds.getArea() < 10) {
039            bounds.extend(bounds.getMax().lat() + 0.0001, bounds.getMax().lon() + 0.0001);
040            bounds.extend(bounds.getMin().lat() - 0.0001, bounds.getMin().lon() - 0.0001);
041        } else if (bounds.getArea() > 10) {
042            JOptionPane.showMessageDialog(Main.parent, tr("It is a big area, it can't be downloaded!"), tr("Warning"), JOptionPane.WARNING_MESSAGE);
043            return;
044        }
045        final Future<?> future = task.download(true, bounds, monitor);
046        Runnable runAfterTask;
047
048        if (osm_obj_id != null) {
049            runAfterTask = new Runnable() {
050                @Override
051                public void run() {
052                    try {
053                        if (osm_obj_id != 0) {
054                            future.get();
055                            //create object
056                            if (future.isDone()) {
057
058                                DataSet dataset = MainApplication.getLayerManager().getEditLayer().data;
059                                Node node = new Node(osm_obj_id);
060                                Relation relation = new Relation(osm_obj_id);
061                                Way way = new Way(osm_obj_id);
062
063                                //create list of objects
064                                List<OsmPrimitive> selection = new ArrayList<>();
065
066                                if (dataset.allPrimitives().contains(node)) {
067                                    selection.add(node);
068                                    dataset.setSelected(selection);
069
070                                } else if (dataset.allPrimitives().contains(way)) {
071                                    selection.add(way);
072                                    dataset.setSelected(selection);
073
074                                } else if (dataset.allPrimitives().contains(relation)) {
075                                    selection.add(relation);
076                                    dataset.setSelected(selection);
077                                }
078                                if (!selection.isEmpty()) {
079                                    AutoScaleAction.autoScale("selection");
080                                }
081                            }
082                        }
083                    } catch (InterruptedException | ExecutionException ex) {
084                        Logger.getLogger(TofixDialog.class.getName()).log(Level.SEVERE, null, ex);
085                    }
086                }
087            };
088            MainApplication.worker.submit(runAfterTask);
089        }
090
091    }
092}
