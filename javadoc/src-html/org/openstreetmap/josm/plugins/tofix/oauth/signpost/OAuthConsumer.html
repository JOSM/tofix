015package org.openstreetmap.josm.plugins.tofix.oauth.signpost;
016
017import java.io.Serializable;
018
019import org.openstreetmap.josm.plugins.tofix.oauth.signpost.exception.OAuthCommunicationException;
020import org.openstreetmap.josm.plugins.tofix.oauth.signpost.exception.OAuthExpectationFailedException;
021import org.openstreetmap.josm.plugins.tofix.oauth.signpost.exception.OAuthMessageSignerException;
022import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpParameters;
023import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpRequest;
024import org.openstreetmap.josm.plugins.tofix.oauth.signpost.signature.AuthorizationHeaderSigningStrategy;
025import org.openstreetmap.josm.plugins.tofix.oauth.signpost.signature.HmacSha1MessageSigner;
026import org.openstreetmap.josm.plugins.tofix.oauth.signpost.signature.OAuthMessageSigner;
027import org.openstreetmap.josm.plugins.tofix.oauth.signpost.signature.QueryStringSigningStrategy;
028import org.openstreetmap.josm.plugins.tofix.oauth.signpost.signature.SigningStrategy;
029
030/**
031 * <p>
032 * Exposes a simple interface to sign HTTP requests using a given OAuth token
033 * and secret. Refer to {@link OAuthProvider} how to retrieve a valid token and
034 * token secret.
035 * </p>
036 * <p>
037 * HTTP messages are signed as follows:
038 * <p>
039 *
040 * <pre>
041 * // exchange the arguments with the actual token/secret pair
042 * OAuthConsumer consumer = new DefaultOAuthConsumer(&quot;1234&quot;, &quot;5678&quot;);
043 *
044 * URL url = new URL(&quot;http://example.com/protected.xml&quot;);
045 * HttpURLConnection request = (HttpURLConnection) url.openConnection();
046 *
047 * consumer.sign(request);
048 *
049 * request.connect();
050 * </pre>
051 *
052 * </p>
053 * </p>
054 *
055 * @author Matthias Kaeppler
056 */
057public interface OAuthConsumer extends Serializable {
058
059    /**
060     * Sets the message signer that should be used to generate the OAuth
061     * signature.
062     *
063     * @param messageSigner
064     *        the signer
065     * @see HmacSha1MessageSigner
066     */
067    public void setMessageSigner(OAuthMessageSigner messageSigner);
068
069    /**
070     * Allows you to add parameters (typically OAuth parameters such as
071     * oauth_callback or oauth_verifier) which will go directly into the signer,
072     * i.e. you don't have to put them into the request first. The consumer's
073     * {@link SigningStrategy} will then take care of writing them to the
074     * correct part of the request before it is sent. This is useful if you want
075     * to pre-set custom OAuth parameters. Note that these parameters are
076     * expected to already be percent encoded -- they will be simply merged
077     * as-is. <b>BE CAREFUL WITH THIS METHOD! Your service provider may decide
078     * to ignore any non-standard OAuth params when computing the signature.</b>
079     *
080     * @param additionalParameters
081     *        the parameters
082     */
083    public void setAdditionalParameters(HttpParameters additionalParameters);
084
085    /**
086     * Defines which strategy should be used to write a signature to an HTTP
087     * request.
088     *
089     * @param signingStrategy
090     *        the strategy
091     * @see AuthorizationHeaderSigningStrategy
092     * @see QueryStringSigningStrategy
093     */
094    public void setSigningStrategy(SigningStrategy signingStrategy);
095
096    /**
097     * <p>
098     * Causes the consumer to always include the oauth_token parameter to be
099     * sent, even if blank. If you're seeing 401s during calls to
100     * {@link OAuthProvider#retrieveRequestToken}, try setting this to true.
101     * </p>
102     *
103     * @param enable
104     *        true or false
105     */
106    public void setSendEmptyTokens(boolean enable);
107
108    /**
109     * Signs the given HTTP request by writing an OAuth signature (and other
110     * required OAuth parameters) to it. Where these parameters are written
111     * depends on the current {@link SigningStrategy}.
112     *
113     * @param request
114     *        the request to sign
115     * @return the request object passed as an argument
116     * @throws OAuthMessageSignerException
117     * @throws OAuthExpectationFailedException
118     * @throws OAuthCommunicationException
119     */
120    public HttpRequest sign(HttpRequest request) throws OAuthMessageSignerException,
121            OAuthExpectationFailedException, OAuthCommunicationException;
122
123    /**
124     * <p>
125     * Signs the given HTTP request by writing an OAuth signature (and other
126     * required OAuth parameters) to it. Where these parameters are written
127     * depends on the current {@link SigningStrategy}.
128     * </p>
129     * This method accepts HTTP library specific request objects; the consumer
130     * implementation must ensure that only those request types are passed which
131     * it supports.
132     *
133     * @param request
134     *        the request to sign
135     * @return the request object passed as an argument
136     * @throws OAuthMessageSignerException
137     * @throws OAuthExpectationFailedException
138     * @throws OAuthCommunicationException
139     */
140    public HttpRequest sign(Object request) throws OAuthMessageSignerException,
141            OAuthExpectationFailedException, OAuthCommunicationException;
142
143    /**
144     * <p>
145     * "Signs" the given URL by appending all OAuth parameters to it which are
146     * required for message signing. The assumed HTTP method is GET.
147     * Essentially, this is equivalent to signing an HTTP GET request, but it
148     * can be useful if your application requires clickable links to protected
149     * resources, i.e. when your application does not have access to the actual
150     * request that is being sent.
151     * </p>
152     *
153     * @param url
154     *        the input URL. May have query parameters.
155     * @return the input URL, with all necessary OAuth parameters attached as a
156     *         query string. Existing query parameters are preserved.
157     * @throws OAuthMessageSignerException
158     * @throws OAuthExpectationFailedException
159     * @throws OAuthCommunicationException
160     */
161    public String sign(String url) throws OAuthMessageSignerException,
162            OAuthExpectationFailedException, OAuthCommunicationException;
163
164    /**
165     * Sets the OAuth token and token secret used for message signing.
166     *
167     * @param token
168     *        the token
169     * @param tokenSecret
170     *        the token secret
171     */
172    public void setTokenWithSecret(String token, String tokenSecret);
173
174    public String getToken();
175
176    public String getTokenSecret();
177
178    public String getConsumerKey();
179
180    public String getConsumerSecret();
181
182    /**
183     * Returns all parameters collected from the HTTP request during message
184     * signing (this means the return value may be NULL before a call to
185     * {@link #sign}), plus all required OAuth parameters that were added
186     * because the request didn't contain them beforehand. In other words, this
187     * is the exact set of parameters that were used for creating the message
188     * signature.
189     *
190     * @return the request parameters used for message signing
191     */
192    public HttpParameters getRequestParameters();
193}
