001package org.openstreetmap.josm.plugins.tofix.oauth.signpost.signature;
002
003import java.io.Serializable;
004
005import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpRequest;
006import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpParameters;
007
008/**
009 * <p>
010 * Defines how an OAuth signature string is written to a request.
011 * </p>
012 * <p>
013 * Unlike {@link OAuthMessageSigner}, which is concerned with <i>how</i> to
014 * generate a signature, this class is concered with <i>where</i> to write it
015 * (e.g. HTTP header or query string).
016 * </p>
017 * 
018 * @author Matthias Kaeppler
019 */
020public interface SigningStrategy extends Serializable {
021
022    /**
023     * Writes an OAuth signature and all remaining required parameters to an
024     * HTTP message.
025     * 
026     * @param signature
027     *        the signature to write
028     * @param request
029     *        the request to sign
030     * @param requestParameters
031     *        the request parameters
032     * @return whatever has been written to the request, e.g. an Authorization
033     *         header field
034     */
035    String writeSignature(String signature, HttpRequest request, HttpParameters requestParameters);
036    
037}
