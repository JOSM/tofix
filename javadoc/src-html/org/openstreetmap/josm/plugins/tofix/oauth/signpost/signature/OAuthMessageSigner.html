015package org.openstreetmap.josm.plugins.tofix.oauth.signpost.signature;
016
017import java.io.IOException;
018import java.io.Serializable;
019import java.util.Base64;
020
021import org.openstreetmap.josm.plugins.tofix.oauth.signpost.exception.OAuthMessageSignerException;
022import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpRequest;
023import org.openstreetmap.josm.plugins.tofix.oauth.signpost.http.HttpParameters;
024
025public abstract class OAuthMessageSigner implements Serializable {
026
027    private static final long serialVersionUID = 4445779788786131202L;
028
029    private String consumerSecret;
030
031    private String tokenSecret;
032
033
034    public abstract String sign(HttpRequest request, HttpParameters requestParameters)
035            throws OAuthMessageSignerException;
036
037    public abstract String getSignatureMethod();
038
039    public String getConsumerSecret() {
040        return consumerSecret;
041    }
042
043    public String getTokenSecret() {
044        return tokenSecret;
045    }
046
047    public void setConsumerSecret(String consumerSecret) {
048        this.consumerSecret = consumerSecret;
049    }
050
051    public void setTokenSecret(String tokenSecret) {
052        this.tokenSecret = tokenSecret;
053    }
054
055    protected byte[] decodeBase64(String s) {
056        return Base64.getDecoder().decode(s);
057    }
058
059    protected String base64Encode(byte[] b) {
060        return Base64.getEncoder().encodeToString(b);
061    }
062
063    private void readObject(java.io.ObjectInputStream stream)
064            throws IOException, ClassNotFoundException {
065        stream.defaultReadObject();
066    }
067}
