084    private String getAuthoriseUrl(OAuthToken requestToken) {
085        //<editor-fold defaultstate="collapsed" desc="comment">
086        StringBuilder sb = new StringBuilder(32);
087
088        // OSM is an OAuth 1.0 provider and JOSM isn't a web app. We just add the oauth request token to
089        // the authorisation request, no callback parameter.
090        //
091        sb.append(oauthProviderParameters.getAuthoriseUrl()).append('?' + OAuth.OAUTH_TOKEN + '=').append(requestToken.getKey());
092        return sb.toString();
093        //</editor-fold>
094    }

096    public void getOAuthToken(SessionId sessionId, OAuthToken requestToken) throws Exception {
097        //<editor-fold defaultstate="collapsed" desc="comment">
098        HttpClient connection = null;
099        try {
100            URL url = new URL(getAuthoriseUrl(requestToken));
101            synchronized (this) {
102                connection = HttpClient.create(url)
103                        .useCache(false)
104                        .setHeader("Cookie", "_osm_session=" + sessionId.getId() + "; _osm_username=" + sessionId.getUserName());
105                connection.connect();
106            }
107            //System.out.println(connection.getResponse().getContentType());
108            sessionId.setToken(extractToken(connection));
109            if (sessionId.getToken() == null) {
110                throw new Exception(tr("OSM website did not return a session cookie in response to ''{0}'',",
111                        url.toString()));
112            }
113        } catch (Exception e) {
114            throw new Exception(e);
115        } finally {
116            synchronized (this) {
117                connection = null;
118            }
119            Logging.info("Finish OAuthToken!!!");
120        }
121        //</editor-fold>
122    }

125        //<editor-fold defaultstate="collapsed" desc="comment">
126        HttpClient connection = null;
127        SessionId sessionId;
128        try {
129            URL url = new URL(oauthProviderParameters.getOsmLoginUrl() + "?cookie_test=true");
130            synchronized (this) {

132                connection.connect();

135            if (sessionId == null) {
136                throw new Exception(tr("OSM website did not return a session cookie in response to ''{0}'',", url.toString()));

139        } catch (Exception e) {
140            throw new Exception(e.getMessage());
141        } finally {
142            synchronized (this) {

144                    connection.disconnect();
145                    connection = null;
146                }
147
148            }
149        }

154    private static String buildPostRequest(Map<String, String> parameters) {
155        //<editor-fold defaultstate="collapsed" desc="comment">
156        StringBuilder sb = new StringBuilder(32);
157
158        for (Iterator<Map.Entry<String, String>> it = parameters.entrySet().iterator(); it.hasNext();) {
159            Map.Entry<String, String> entry = it.next();
160            String value = entry.getValue();
161            value = (value == null) ? "" : value;
162            sb.append(entry.getKey()).append('=').append(Utils.encodeUrl(value));
163            if (it.hasNext()) {
164                sb.append('&');
165            }
166        }
167        return sb.toString();
168        //</editor-fold>
169    }

171    public SessionId login(String userName, String password) throws Exception {
172        //<editor-fold defaultstate="collapsed" desc="comment">
173        HttpClient connection;
174        SessionId sessionId=null;
175        try {
176            URL url = new URL(oauthProviderParameters.getOsmLoginUrl());
177            connection = HttpClient.create(url, "POST").useCache(false);
178            sessionId = getOsmWebsiteSessionId(userName);
179            Map<String, String> parameters = new HashMap<>();
180            parameters.put("username", userName);
181            parameters.put("password", password);
182            parameters.put("referer", "/");
183            parameters.put("commit", "Login");
184            parameters.put("authenticity_token", sessionId.getToken());
185            connection.setRequestBody(buildPostRequest(parameters).getBytes(StandardCharsets.UTF_8));
186
187            connection.setHeader("Content-Type", "application/x-www-form-urlencoded");
188            connection.setHeader("Cookie", "_osm_session=" + sessionId.getId());
189            // make sure we can catch 302 Moved Temporarily below
190            connection.setMaxRedirects(-1);
191
192            synchronized (this) {
193                connection.connect();
194            }
195
196            // after a successful login the OSM website sends a redirect to a follow up page. Everything
197            // else, including a 200 OK, is a failed login. A 200 OK is replied if the login form with
198            // an error page is sent to back to the user.
199            //
200            int retCode = connection.getResponse().getResponseCode();
201            if (retCode != HttpURLConnection.HTTP_MOVED_TEMP) {
202                throw new Exception(tr("Failed to authenticate user ''{0}'' with password ''***'' as OAuth user",
203                        userName));
204            }
205        } catch (Exception e) {
206            throw new Exception(e);
207        } finally {
208            synchronized (this) {
209                connection = null;
210            }
211        }
212        return sessionId;
213        //</editor-fold>
214    }

236    private SessionId extractOsmSession(HttpClient connection) throws IOException, URISyntaxException {
237        //<editor-fold defaultstate="collapsed" desc="comment">
238        // response headers might not contain the cookie, see #12584
239        final List<String> setCookies = CookieHandler.getDefault()
240                .get(connection.getURL().toURI(), Collections.<String, List<String>>emptyMap())
241                .get("Cookie");
242        if (setCookies == null) {
243            Logging.warn("No 'Set-Cookie' in response header!");
244            return null;
245        }
246
247        for (String setCookie : setCookies) {
248            String[] kvPairs = setCookie.split(";");
249            if (kvPairs.length == 0) {
250                continue;
251            }
252            for (String kvPair : kvPairs) {
253                kvPair = kvPair.trim();
254                String[] kv = kvPair.split("=");
255                if (kv.length != 2) {
256                    continue;
257                }
258                if ("_osm_session".equals(kv[0])) {
259                    // osm session cookie found
260                    String token = extractToken(connection);
261                    if (token == null) {
262                        return null;
263                    }
264                    SessionId si = new SessionId();
265                    si.setId(kv[1]);
266                    si.setToken(token);
267                    return si;
268                }
269            }
270        }
271        Logging.warn("No suitable 'Set-Cookie' in response header found! {0}", setCookies);
272        return null;
273        //</editor-fold>
274    }

276    private String extractToken(HttpClient connection) {
277        //<editor-fold defaultstate="collapsed" desc="comment">
278        try (BufferedReader r = connection.getResponse().getContentReader()) {
279            String c;
280            Pattern p = Pattern.compile(".*authenticity_token.*value=\"([^\"]+)\".*");
281            while ((c = r.readLine()) != null) {
282                Matcher m = p.matcher(c);
283                if (m.find()) {
284                    return m.group(1);
285                }
286            }
287        } catch (IOException e) {
288            Logging.error(e);
289            return null;
290        }
291        Logging.warn("No authenticity_token found in response!");
292        return null;
293        //</editor-fold>
294    }

296    private void sendAuthorisationRequest(SessionId sessionId, OAuthToken requestToken, OsmPrivileges privileges)throws Exception {
297        //<editor-fold defaultstate="collapsed" desc="comment">
298        HttpClient connection;
299        Map<String, String> parameters = new HashMap<>();
300        getOAuthToken(sessionId, requestToken);
301        parameters.put("oauth_token", requestToken.getKey());
302        parameters.put("oauth_callback", "");
303        parameters.put("authenticity_token", sessionId.getToken());
304        if (privileges.isAllowWriteApi()) {
305            parameters.put("allow_write_api", "yes");
306        }
307        if (privileges.isAllowWriteGpx()) {
308            parameters.put("allow_write_gpx", "yes");
309        }
310        if (privileges.isAllowReadGpx()) {
311            parameters.put("allow_read_gpx", "yes");
312        }
313        if (privileges.isAllowWritePrefs()) {
314            parameters.put("allow_write_prefs", "yes");
315        }
316        if (privileges.isAllowReadPrefs()) {
317            parameters.put("allow_read_prefs", "yes");
318        }
319        if (privileges.isAllowModifyNotes()) {
320            parameters.put("allow_write_notes", "yes");
321        }
322
323        parameters.put("commit", "Save changes");
324
325        String request = buildPostRequest(parameters);
326        try {
327            URL url = new URL(oauthProviderParameters.getAuthoriseUrl());
328            connection = HttpClient.create(url, "POST").useCache(false);
329            connection.setHeader("Content-Type", "application/x-www-form-urlencoded");
330            connection.setHeader("Cookie", "_osm_session=" + sessionId.getId() + "; _osm_username=" + sessionId.getUserName());
331            connection.setMaxRedirects(-1);
332            connection.setRequestBody(request.getBytes(StandardCharsets.UTF_8));
333
334            synchronized (this) {
335                connection.connect();
336            }
337
338            int retCode = connection.getResponse().getResponseCode();
339            if (retCode != HttpURLConnection.HTTP_OK)
340                throw new Exception(tr("Failed to authorize OAuth request  ''{0}''", requestToken.getKey()));
341        } catch (IOException e) {
342            throw new Exception(e);
343        } finally {
344            synchronized (this) {
345                connection = null;
346            }
347        }
348        //</editor-fold>
349    }

351    public SessionId authorise(String userName, String password, OsmPrivileges privileges)throws Exception {
352        //<editor-fold defaultstate="collapsed" desc="comment">
353        OAuthToken requestToken = getRequestToken();
354        CheckParameterUtil.ensureParameterNotNull(requestToken, "requestToken");
355        CheckParameterUtil.ensureParameterNotNull(userName, "userName");
356        CheckParameterUtil.ensureParameterNotNull(password, "password");
357        CheckParameterUtil.ensureParameterNotNull(privileges, "privileges");
358        SessionId sessionId;
359        try {
360            Logging.info(tr("Authorizing OAuth Request token ''{0}'' at the OSM website ...", requestToken.getKey()));
361            Logging.info(tr("Initializing a session at the OSM website...")); 
362            sessionId = login(userName, password);
363            Logging.info(tr("Authenticating the session for user ''{0}''...", userName));
364            Logging.info(tr("Authorizing request token ''{0}''...", requestToken.getKey()));
365            sendAuthorisationRequest(sessionId, requestToken, privileges);            
366            Logging.info(tr("Logging out session ''{0}''...", sessionId));
367            logout(sessionId);            
368        } catch (Exception e) {            
369            throw e;
370        } finally {
371            Logging.info("Finish Authorise!!!");
372        }
373        return sessionId;
374        //</editor-fold>
375    }
