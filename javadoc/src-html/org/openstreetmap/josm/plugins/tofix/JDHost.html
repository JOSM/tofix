144    private void runTest() {
145        if (Config.isURL(jTextField1.getText())) {
146            if (Config.preferences(Config.GET, new String[]{"tofix-server.host"}, Config.getPluginPreferencesFile().getAbsolutePath()) != null) {
147                Config.preferences(Config.UPDATE, new String[]{"tofix-server.host", jTextField1.getText()}, Config.getPluginPreferencesFile().getAbsolutePath());
148            } else {
149                Config.preferences(Config.ADD, new String[]{"tofix-server.host", jTextField1.getText()}, Config.getPluginPreferencesFile().getAbsolutePath());
150            }
151            if (Status.serverStatus()) {
152                jLabel2.setText(tr("Successful test with server!!!"));
153                jLabel2.setForeground(new Color(0,153,0));//GREEN
154            } else {
155                jLabel2.setText(tr("Server Not Found!!!"));
156                jLabel2.setForeground(Color.RED);
157            }
158        } else {
159            jLabel2.setText(tr("Incorrect URL"));
160            jLabel2.setForeground(Color.RED);
161        }
162    }
